CREATE DATABASE BIBLIOTECA;
USE BIBLIOTECA;

CREATE TABLE FUNCIONARIOS
(
CPF NUMERIC(11) PRIMARY KEY,
NOME VARCHAR(70),
ENDERECO VARCHAR(60),
TELEFONE NUMERIC(15),
SALARIO DECIMAL(7,3),
FUNCAO VARCHAR(25)
);

CREATE TABLE EDITORAS
(
CODIGO INT(20) PRIMARY KEY,
NOME VARCHAR(70),
ENDERECO VARCHAR(60),
CONTATO NUMERIC(15)
);


CREATE TABLE USUARIOS 
(
CPF NUMERIC(11) PRIMARY KEY,
NOME VARCHAR(70),
ENDERECO VARCHAR(60),
TELEFONE NUMERIC(15)
);

CREATE TABLE LIVROS
(
NUMERO NUMERIC(25) PRIMARY KEY,
TITULO VARCHAR(110),
GENERO VARCHAR(25),
EDICAO INT(5),
ANO_PUBLICACAO YEAR,
CPF_FUNCIONARIO NUMERIC(11) REFERENCES FUNCIONARIOS,
CODIGO_EDITORA NUERIC(20) REFERENCES EDITORAS,
CPF_USUARIORETIRAR NUMERIC(11) REFERENCES USUARIOS,
CPF_USUARIORESERVAR NUMERIC(11) REFERENCES USUARIOS
);


CREATE TABLE AUTORES
(
CODIGO NUMERIC(25) PRIMARY KEY,
NOME VARCHAR(70),
NACIONALIDADE VARCHAR(20)
);

CREATE TABLE LIVROS_AUTORES
(
NUMERO_LIVRO NUMERIC(25) REFERENCES LIVROS,
CODIGO_AUTOR NUMERIC(25) REFERENCES AUTORES
);
